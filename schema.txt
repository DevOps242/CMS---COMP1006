CREATE TABLE IF NOT EXISTS CMSUsers (
	userID INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    userFirstName VARCHAR(100) NOT NULL,
    userLastName VARCHAR(100) NOT NULL,
    userEmail VARCHAR(150) NOT NULL,
    userPassword VARCHAR(255) NOT NULL,
    userGUID VARCHAR(255) NOT NULL,
    userStatus ENUM('active', 'inactive') NOT NULL,
    userCreatedOn TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    userModifiedOn TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX (userGUID)
) AUTO_INCREMENT = 10000;


CREATE TABLE IF NOT EXISTS CMSPages (
	pageID INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    pageName VARCHAR(255) NOT NULL,
    pageTitle VARCHAR(255) NOT NULL,
    pageImg VARCHAR(255) NULL,
    pageStatus ENUM('active', 'inactive') NOT NULL,
    userGUID VARCHAR(255) NOT NULL,
    pageCreatedOn TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    pageModifiedOn TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX (userGUID),
    FOREIGN KEY (userGUID) REFERENCES CMSUsers (userGUID)
) AUTO_INCREMENT = 10000;


CREATE TABLE IF NOT EXISTS CMSLogos (
    logoID INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    logoName VARCHAR(255) NOT NULL,
    logoTitle VARCHAR(255) NOT NULL,
    logoImg VARCHAR(255) NULL,
    logoStatus ENUM('active', 'inactive') NOT NULL,
    logoCreatedOn TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    logoModifiedOn TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) AUTO_INCREMENT = 10000;
